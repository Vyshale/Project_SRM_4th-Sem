<script>
  // ============================================================
  // Screen routing
  // ============================================================
  let currentRole = 'Customer';

  function show(screenName) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById('screen-' + screenName).classList.add('active');
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  function pickRole(el, role) {
    el.parentElement.querySelectorAll('.role-card').forEach(c => c.classList.remove('active'));
    el.classList.add('active');
    if (role) currentRole = role;
  }

  function doLogin() {
    try {
      const roleEl = document.getElementById('sb-user-role');
      if (roleEl) roleEl.textContent = currentRole || 'Customer';
    } catch (e) { /* ignore */ }
    show('app');
    goto('dashboard');
    return false;
  }

  function doLogout() {
    show('login');
  }

  // ============================================================
  // Page rendering inside the app shell
  // ============================================================
  const PAGE_TITLES = {
    dashboard: 'Dashboard',
    water: 'Water Tank',
    solar: 'Solar Energy',
    filter: 'Water Filter',
    service: 'Service History',
    complaints: 'Complaints',
    payments: 'Payments',
    devices: 'My Devices'
  };

  function goto(pageName) {
    const tpl = document.getElementById('tpl-' + pageName);
    if (!tpl) return;
    const content = document.getElementById('page-content');
    content.innerHTML = '';
    content.appendChild(tpl.content.cloneNode(true));

    // active nav
    document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
    const activeLink = document.querySelector('.nav-link[data-page="' + pageName + '"]');
    if (activeLink) activeLink.classList.add('active');

    // crumb
    document.getElementById('crumb-page').textContent = PAGE_TITLES[pageName] || pageName;

    // scroll to top of main
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  // ============================================================
  // Misc widgets
  // ============================================================
  function toggleSev(el, level) {
    el.parentElement.querySelectorAll('.sev').forEach(s => s.classList.remove('active'));
    el.classList.add('active');
  }

  function pickPay(el) {
    el.parentElement.querySelectorAll('.pay-method').forEach(p => p.classList.remove('active'));
    el.classList.add('active');
  }
</script>
